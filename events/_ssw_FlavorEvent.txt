	id = flavor.1
	title = flavor.1.t
	desc = flavor.1.d
	picture = GFX_1950_WorldCup.dds
	
	major = yes
	fire_only_once = yes
	
	trigger = {
		date > 1950.7.16
		country_exists = BRA
		country_exists = URG
		BRA = { has_war = no }
		URG = { has_civil_war = no }
	}

		days = 1
	}
	
	
	option = {
		name = flavor.1.a
		trigger = { 
			NOT = { TAG = URG }
		}
	}
	option = {
		name = flavor.1.b
		trigger = { TAG = URG }
	}
}
