add_namespace = ssw_ussr
country_event = {
	id = ssw_ussr.0
	hidden = yes
	trigger = {
		tag = SOV
		has_war = no
		country_exists = SOV
		NOT = { country_exists = RUS }
	}
	fire_only_once = yes
	mean_time_to_happen = { days = 1 }
	immediate = {
		every_state = {
			limit = { is_core_of = RUS }
			add_core_of = SOV
			remove_core_of = RUS
		}
		every_state = {
			limit = {
				OR = {
					is_core_of = MOS
					is_core_of = UKR
					is_core_of = CCS
					is_core_of = BAL
					is_core_of = PET
					is_core_of = SIB
				}
				NOT = { is_core_of = SOV }
			}
			add_core_of = SOV
		}
		if = {
			limit = { SOV = { has_cosmetic_tag = SOV_provisional_government } }
			SOV = { 
				drop_cosmetic_tag = yes
				set_cosmetic_tag = SOV_republics_union
			}
		}
		set_global_flag = russian_war_end
	}
	option = { name = ssw_ussr.0.a }
}
