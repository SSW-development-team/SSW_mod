    id = FNO_event.1
	title = FNO_event.1.t
	desc = FNO_event.1.d
	picture = GFX_report_event_french_resistance_02
	fire_only_once = yes
	
	trigger = {
		tag = SWE
		has_global_flag = GER_civil_war
	}
	
	option = {
		name = FNO_event.1.a
		ai_chance = { base = 100 }
		
		SWE = {
			add_political_power = -15
			send_equipment = {
				equipment = infantry_equipment
				amount = 150
				target = FNO
			}
		}
		hidden_effect = {
			FNO = {
				add_opinion_modifier = {
					target = SWE
					modifier = SSW_support_civil_war
				}
			}
		}
	}
	
	option = {
		name = FNO_event.1.b
		ai_chance = { base = 0 }
	}
}
