		}
	}

}

##コロリョフの勝利
country_event = {
	id = sswrfs.4
	title = sswrfs.4.title
	desc = sswrfs.4.desc
	picture = GFX_event_test
	fire_only_once = yes
	is_triggered_only = yes

	trigger = {
		tag = RBL

			political_strife_kor > 120
		}
	}

		remove_ideas = RFS_political_strife
		add_political_power = 100

			set_country_flag = win_kor
		}
	}
}
