    SER_martial_law_time_limit = {
        icon = generic_speech #アイコン
		ai_will_do = {
			factor = 1 #AIが踏む確率
		}
		allowed = {
			original_tag = SER #踏めるようになる条件
		}
		visible = { #見えるようになる条件
		}
		fire_only_once = yes #何回も使えるか
		days_remove = 10 #決定したときにディシジョンを取得しきれる日数
		remove_effect = {
            if = {
                limit = {
                    OR = {
                        check_variable = {
                            var = Martial_Law_in_Serbia_var
                            value = 3
                            compare = equals
                        }
                        check_variable = {
                            var = Martial_Law_in_Morava_var
                            value = 3
                            compare = equals
                        }
                        check_variable = {
                            var = Martial_Law_in_Kosovo_var
                            value = 3
                            compare = equals
                        }
                    }
                }
                add_political_power = -300
                add_stability = -0.50
                add_war_support = -0.50
                swap_ideas = {
                    remove_idea = SER_idea_20
                    add_idea = my_idea_26
                }
            }
            if = {
                limit = {
                    OR = {
                        check_variable = {
                            var = Martial_Law_in_Serbia_var
                            value = 2
                            compare = equals
                        }
                        check_variable = {
                            var = Martial_Law_in_Morava_var
                            value = 2
                            compare = equals
                        }
                        check_variable = {
                            var = Martial_Law_in_Kosovo_var
                            value = 2
                            compare = equals
                        }
                    }
                }
                add_political_power = -150
                add_stability = -0.25
                add_war_support = -0.25
                swap_ideas = {
                    remove_idea = SER_idea_20
                    add_idea = my_idea_27
                }
            }
            if = {
                limit = {
                    OR = {
                        check_variable = {
                            var = Martial_Law_in_Serbia_var
                            value = 1
                            compare = equals
                        }
                        check_variable = {
                            var = Martial_Law_in_Morava_var
                            value = 1
                            compare = equals
                        }
                        check_variable = {
                            var = Martial_Law_in_Kosovo_var
                            value = 1
                            compare = equals
                        }
                    }
                }
                add_political_power = -100
                add_stability = -0.10
                add_war_support = -0.10
                swap_ideas = {
                    remove_idea = SER_idea_20
                    add_idea = my_idea_28
                }
            }
            if = {
                limit = {
                    AND = {
                        check_variable = {
                            var = Martial_Law_in_Serbia_var
                            value = 0
                            compare = equals
                        }
                        check_variable = {
                            var = Martial_Law_in_Morava_var
                            value = 0
                            compare = equals
                        }
                        check_variable = {
                            var = Martial_Law_in_Kosovo_var
                            value = 0
                            compare = equals
                        }
                    }
                }
                add_political_power = 30
                add_stability = 0.10
                add_war_support = 0.10
                remove_ideas = SER_idea_20
                add_timed_idea = {
                    idea = SER_idea_29
                    days = 200
                }
            }
		}
    }
