d_SSW_Cheat_Lists = {
	log = "Randomize_GCW"
	log = "start_GCW"
}

d_set_goering = {
	if = {
		limit = {
			is_debug = yes
			check_variable = { args^num = 1 }
		}
		for_each_scope_loop = {
			array = args
			PREV = {
				set_variable = { Current_Goering_Influence = args^0 }
			}
		}
		log = "[?PREV.GetName]"
		log = "[?value]"
		log = "[?args^0]"
		log = "[?FROM.GetName]"
		log = "----------------------"
	}
	else = {
		log = "Command was used in wrong scope!"
	}
}

d_disband_all = {
	set_variable = { disband_fraction = args^0 }
	disband_units_fraction = yes
	clear_variable = disband_fraction
}

d_Randomize_GCW = {
	FROM = {
		every_owned_state = {
			set_variable = { Current_Goering_Influence = random }
			set_variable = { Current_Himmler_Influence = random }
			set_variable = { Current_Goebbels_Influence = random }
			set_variable = { Current_Bormann_Influence = random }
		}
	}
}

d_start_GCW = {
	FROM = {
		every_owned_state = {
			if = {
				limit = { Goering_is_Best = yes }
				HGR ={
					transfer_state = PREV
				}
				Check_load_oob = yes
			}
			if = {
				limit = { Himmler_is_Best = yes }
				HLH ={
					transfer_state = PREV
				}
			}
			if = {
				limit = { Goebbels_is_Best = yes }
				GER ={
					transfer_state = PREV
				}
			}
			if = {
				limit = { Bormann_is_Best = yes }
				MBR ={
					transfer_state = PREV
				}
			}
		}
	}
	HGR = {
		declare_war_on = {
			target = HLH
			type = annex_everything
		}
		declare_war_on = {
			target = GER
			type = annex_everything
		}
		declare_war_on = {
			target = MBR
			type = annex_everything
		}
	}
	HLH = {
		declare_war_on = {
			target = GER
			type = annex_everything
		}
		declare_war_on = {
			target = MBR
			type = annex_everything
		}
	}
	GER = {
		declare_war_on = {
			target = MBR
			type = annex_everything
		}
	}
}

d_load_oob_infantry = {
	create_unit = {
		division = "division_template = \"Infanterie-Division\" start_experience_factor = 0.1  start_equipment_factor = 0.5" owner = FROM
	}
	log = "Created"
}
