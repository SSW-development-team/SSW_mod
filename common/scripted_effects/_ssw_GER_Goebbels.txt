	every_controlled_state = {
		add_to_temp_variable = { temp1 = state_population_k }
	}
	set_variable = { All_Pop = temp1 }
}

GDP_Culculate_Average_GDP = {
	set_temp_variable = { temp1 = All_GDP }
	divide_temp_variable = { temp1 = All_Pop }
	set_variable = { Average_GDP = temp1 }
}
GDP_Culculate_Infra_worker = {
	set_temp_variable = { temp4 = All_Pop }
	every_controlled_state = {
		add_to_temp_variable = { temp4 = building_level@infrastructure }
		add_to_temp_variable = { temp4 = building_level@synthetic_refinery }
		add_to_temp_variable = { temp4 = building_level@fuel_silo }
		add_to_temp_variable = { temp4 = building_level@radar_station }
		add_to_temp_variable = { temp4 = building_level@nuclear_reactor }
	}
	divide_temp_variable = { temp4 = num_controlled_states }
	set_variable = { All_Infra = temp4 }
}
