scripted_gui = {	
	ssw_research_atomic_panel = {
		context_type = player_context
		parent_window_token = top_bar
		window_name = "ssw_research_atomic_panel"

		visible = {
			is_ai = no
		}

		effects = {
			button_open_atomic_gui_click = {
				if = {
					limit = { has_variable = ssw_atomic_gui }
					clear_variable = ssw_atomic_gui

				}
				else = {
					set_variable = { ssw_atomic_gui = 1 }
				}
			}
		}
	}
	ssw_research_atomic_window = {
		context_type = player_context
		parent_window_token = top_bar
		window_name = "ssw_research_atomic_window"

		visible = {
			has_variable = ssw_atomic_gui
		}

		effects = {
			button_open_atomic_gui_click = {
			}
			ssw_research_atomic_production_uranium_ic_up_click = {
				if = {
					limit = {
						check_variable = { ssw_atomic_amount_ic_uranium < num_of_civilian_factories }
					}
					add_to_variable = { ssw_atomic_amount_ic_uranium = 1 }
					calculate_value_of_uranium = yes
				}
				else = {
				}
			}
			ssw_research_atomic_production_uranium_ic_down_click = {
				if = {
					limit = {
						check_variable = { ssw_atomic_amount_ic_uranium > 0 }
					}
					add_to_variable = { ssw_atomic_amount_ic_uranium = -1 }
					calculate_value_of_uranium = yes

				}
				else = {
				}
			}
			ssw_research_atomic_production_uranium_nukeplant_up_click = {
				if = {
					limit = {
						check_variable = { ssw_atomic_amount_nukeplant_uranium < num_of_nuclear_reactor }
					}
					add_to_variable = { ssw_atomic_amount_nukeplant_uranium = 1 }
					calculate_value_of_uranium = yes

				}
				else = {
				}
			}
			ssw_research_atomic_production_uranium_nukeplant_down_click = {
				if = {
					limit = {
						check_variable = { ssw_atomic_amount_nukeplant_uranium > 0 }
					}
					add_to_variable = { ssw_atomic_amount_nukeplant_uranium = -1 }
					calculate_value_of_uranium = yes

				}
				else = {
				}
			}

			test_click = {
				add_to_variable = { uranium_value = 1 }
				calculate_value_of_uranium = yes
			}

			ssw_research_atomic_bomb_buttom_click = {
				if = {
					limit = { 	check_variable = { ssw_atomic_gui = 2 }
 }
					set_variable = { ssw_atomic_gui = 1 }
				}
				else = {
					set_variable = { ssw_atomic_gui = 2 }
				}

			}


		}
		properties = {
			ssw_research_atomic_ber = {
				frame = uranium_late
			}
		}
		triggers = {
		}
	}
	ssw_research_atomic__bomb_panel = {
		context_type = player_context
		parent_window_token = top_bar
		window_name = "ssw_research_atomic_bomb_window"

		visible = {
			check_variable = { ssw_atomic_gui = 2 }
		}

		effects = {
			button_open_atomic_gui_click = {
			}
		}
	}

}
